CREATE TABLE Country (
  area_code INT PRIMARY KEY NOT NULL UNIQUE,
  country_name VARCHAR
);
DROP TABLE Country;
CREATE TABLE Item (
  item_code INT PRIMARY KEY NOT NULL UNIQUE,
  area_code INT,
  item_name VARCHAR,
  FOREIGN KEY (area_code) REFERENCES Country (area_code)
);
DROP TABLE Item;

CREATE TABLE Element (
  element_code INT PRIMARY KEY NOT NULL UNIQUE,
  area_code INT,
  element_name VARCHAR,
  FOREIGN KEY (area_code) REFERENCES Country (area_code)
);

DROP TABLE Element;

CREATE TABLE Emission (
  area_code INT NOT NULL,
  item_code INT NOT NULL,
  element_code INT NOT NULL,
  emission_year CHAR,
  emission_value FLOAT,
  FOREIGN KEY (area_code) REFERENCES Country (area_code),
  FOREIGN KEY (item_code) REFERENCES Item (item_code),
  FOREIGN KEY (element_code) REFERENCES Element (element_code),
  PRIMARY KEY (area_code, item_code, element_code)
);

DROP TABLE Emission;
